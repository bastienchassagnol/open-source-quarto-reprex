%%%%%% Chapter, title and paragraph options %%%%%%
\usepackage{titlesec} %To change chapter and section title default formats
\usepackage{xhfill} % To add a horizontal rule to the chapter and section titles
\definecolor{rulecolor_dark}{HTML}{1D2A3B}
% Chapter title format
\titleformat{\chapter}[display]
{\filcenter}{\mbox{}\xrfill[0.4ex]{3pt}[rulecolor_dark]\textsc{\large\enspace\chaptername \thechapter}\enspace\xrfill[0.4ex]{3pt}[rulecolor_dark]\mbox{}}{0.3ex} {{\color{rulecolor_dark}\titlerule[1pt]}\vskip3ex\huge\bfseries}[\medskip{\color{rulecolor_dark}\titlerule[1pt]}]
% Section title format
\titleformat{\section}
  {\sffamily\large\bfseries}
  {\thesection}{1ex}{}
  [\vspace{-0.5ex}{\color{lightgray}\titlerule[1pt]}]
\titlespacing*{\section}{0pt}{2ex}{2ex}
% Subsection title format
\titleformat{\subsection}
  {\sffamily\bfseries}
  {\thesubsection}{1em}{}
\titlespacing*{\subsection}{0pt}{1.5ex}{-0.5ex}
